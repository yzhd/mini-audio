mini-audio.js

Api

audio.play() audio.pause() audio.playPause() audio.musicPlay(0) audio.musicPlay(1) audio.setMusic(index) audio.changeMode() audio.showMenu()

Usage

1. Include the audio.js file
<div id="player"></div>
<script src="audio.js"></script>
2. Initialise the audio player
var audio = new Audio({
  element: document.getElementById('player'),  //渲染的mod节点，默认是 #player
  autoplay: true,                             //是否自动播放,默认false
  menu: true,                                  //是否显示菜单
  path: "",                                    //音频文件根目录，模式是root
  music: [{
    url: "",                                   //音乐的Url地址
    title: "",                                 //音乐标题
    author: "",                                //歌手
    img: ""                                    //音乐图片
  }]
});
